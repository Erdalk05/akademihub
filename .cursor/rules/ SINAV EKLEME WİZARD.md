```typescript

// âš ï¸ Ã–NEMLÄ°: Bu iÅŸlem SERVER-SIDE yapÄ±lmalÄ± (Edge Function veya API Route)
// Ã‡Ã¼nkÃ¼ Ã§ok fazla kayÄ±t iÅŸleniyor ve client timeout olabilir

interface Adim5Data {
  sinavId: string;
  yayinSecenegi: 'taslak' | 'hemen' | 'zamanli';
  zamanliYayinTarihi?: Date;
}

async function adim5Yayinla(data: Adim5Data): Promise<void> {
  // Bu fonksiyon /api/admin/exam-analytics/exams/[id]/publish endpoint'inde Ã§alÄ±ÅŸÄ±r
  
  const supabase = createServiceRoleClient(); // Service role gerekli

  // 1. SINAV BÄ°LGÄ°LERÄ°NÄ° AL
  const { data: sinav } = await supabase
    .from('ea_sinavlar')
    .select(`
      *,
      ea_sinav_dersler(*),
      ea_cevap_anahtarlari(*),
      ea_katilimcilar(
        *,
        ea_katilimci_cevaplar(*)
      )
    `)
    .eq('id', data.sinavId)
    .single();

  // 2. DURUMU HESAPLANIYOR YAP
  await supabase
    .from('ea_sinavlar')
    .update({ durum: 'hesaplaniyor' })
    .eq('id', data.sinavId);

  // 3. HER KATILIMCI Ä°Ã‡Ä°N HESAPLA
  const dersSonuclari: any[] = [];
  const genelSonuclar: any[] = [];

  for (const katilimci of sinav.ea_katilimcilar) {
    let toplamDogru = 0, toplamYanlis = 0, toplamBos = 0;
    let sozelDogru = 0, sozelYanlis = 0;
    let sayisalDogru = 0, sayisalYanlis = 0;

    // Her ders iÃ§in hesapla
    for (const sinvDers of sinav.ea_sinav_dersler) {
      const cevapAnahtari = sinav.ea_cevap_anahtarlari
        .find(ca => ca.ders_id === sinvDers.ders_id);
      
      const katilimciCevap = katilimci.ea_katilimci_cevaplar
        .find(kc => kc.ders_id === sinvDers.ders_id);

      if (!cevapAnahtari || !katilimciCevap) continue;

      // DoÄŸru/YanlÄ±ÅŸ/BoÅŸ hesapla
      let dogru = 0, yanlis = 0, bos = 0;
      
      for (let i = 0; i < cevapAnahtari.cevap_dizisi.length; i++) {
        const dogruCevap = cevapAnahtari.cevap_dizisi[i];
        const verilenCevap = katilimciCevap.cevap_dizisi[i] || ' ';
        
        if (verilenCevap === ' ' || verilenCevap === '-' || verilenCevap === '*') {
          bos++;
        } else if (verilenCevap === dogruCevap) {
          dogru++;
        } else {
          yanlis++;
        }
      }

      // Net hesapla
      const net = dogru - (yanlis * sinav.yanlis_katsayi);
      const basariYuzdesi = (dogru / sinvDers.soru_sayisi) * 100;

      // Ders sonucu kaydet
      dersSonuclari.push({
        katilimci_id: katilimci.id,
        ders_id: sinvDers.ders_id,
        soru_sayisi: sinvDers.soru_sayisi,
        dogru,
        yanlis,
        bos,
        net: parseFloat(net.toFixed(2)),
        basari_yuzdesi: parseFloat(basariYuzdesi.toFixed(2)),
      });

      // ToplamlarÄ± gÃ¼ncelle
      toplamDogru += dogru;
      toplamYanlis += yanlis;
      toplamBos += bos;

      // SÃ¶zel/SayÄ±sal ayrÄ±mÄ± (LGS iÃ§in)
      if (['TUR', 'INK', 'DIN', 'ING'].includes(sinvDers.ders_kodu)) {
        sozelDogru += dogru;
        sozelYanlis += yanlis;
      } else {
        sayisalDogru += dogru;
        sayisalYanlis += yanlis;
      }
    }

    // Genel sonuÃ§
    const toplamNet = toplamDogru - (toplamYanlis * sinav.yanlis_katsayi);
    const sozelNet = sozelDogru - (sozelYanlis * sinav.yanlis_katsayi);
    const sayisalNet = sayisalDogru - (sayisalYanlis * sinav.yanlis_katsayi);
    
    // LGS puan hesabÄ± (Ã¶rnek formÃ¼l)
    const lgsPuan = 100 + (toplamNet * 3.333);

    genelSonuclar.push({
      sinav_id: data.sinavId,
      katilimci_id: katilimci.id,
      toplam_soru: sinav.toplam_soru,
      toplam_dogru: toplamDogru,
      toplam_yanlis: toplamYanlis,
      toplam_bos: toplamBos,
      toplam_net: parseFloat(toplamNet.toFixed(2)),
      basari_yuzdesi: parseFloat(((toplamDogru / sinav.toplam_soru) * 100).toFixed(2)),
      // SÃ¶zel/SayÄ±sal
      sozel_net: parseFloat(sozelNet.toFixed(2)),
      sozel_dogru: sozelDogru,
      sozel_yanlis: sozelYanlis,
      sayisal_net: parseFloat(sayisalNet.toFixed(2)),
      sayisal_dogru: sayisalDogru,
      sayisal_yanlis: sayisalYanlis,
      // Puan
      lgs_puan: parseFloat(lgsPuan.toFixed(3)),
    });
  }

  // 4. DERS SONUÃ‡LARINI BULK INSERT
  await supabase
    .from('ea_ders_sonuclari')
    .insert(dersSonuclari);

  // 5. GENEL SONUÃ‡LARI BULK INSERT
  await supabase
    .from('ea_sonuclar')
    .insert(genelSonuclar);

  // 6. SIRALAMA HESAPLA
  // LGS puanÄ±na gÃ¶re sÄ±rala, aynÄ± puanda toplam_net'e gÃ¶re
  const { data: siraliSonuclar } = await supabase
    .from('ea_sonuclar')
    .select('id, lgs_puan, toplam_net')
    .eq('sinav_id', data.sinavId)
    .order('lgs_puan', { ascending: false })
    .order('toplam_net', { ascending: false });

  // SÄ±ra numaralarÄ±nÄ± gÃ¼ncelle
  for (let i = 0; i < siraliSonuclar.length; i++) {
    await supabase
      .from('ea_sonuclar')
      .update({ 
        sira: i + 1,
        yuzdelik: parseFloat((((siraliSonuclar.length - i) / siraliSonuclar.length) * 100).toFixed(2)),
      })
      .eq('id', siraliSonuclar[i].id);
  }

  // 7. SINIFI VE ÅUBEYÄ° Ä°Ã‡Ä°NDE SIRALAMA (Opsiyonel)
  // ... (benzer mantÄ±k, sÄ±nÄ±f/ÅŸube bazlÄ±)

  // 8. SINAV DURUMUNU GÃœNCELLE
  await supabase
    .from('ea_sinavlar')

      durum: data.yayinSecenegi === 'zamanli' ? 'zamanli_yayin' : 'yayinlandi',
      yayinlanma_tarihi: data.yayinSecenegi === 'hemen' 
        ? new Date().toISOString() 
        : data.zamanliYayinTarihi?.toISOString(),
      hesaplama_tarihi: new Date().toISOString(),
```
